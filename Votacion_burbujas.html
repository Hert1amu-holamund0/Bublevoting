  <script>

    // Handle voting bubble selection

    function toggleVote(bubble) {

      bubble.classList.toggle('selected');

    }


    // Handle category size increase on double click

    function increaseCategorySize(categoryElement) {

      const currentSize = parseInt(window.getComputedStyle(categoryElement).fontSize);

      categoryElement.style.fontSize = `${currentSize + 2}px`;

    }


    // Add a new bubble for a custom response

    function addNewBubble() {

      const newResponse = prompt("Enter your new response:");

      if (newResponse) {

        const bubbleContainer = document.querySelector('.bubble-container');

        const newBubble = document.createElement('div');

        newBubble.classList.add('bubble');

        newBubble.textContent = newResponse;

        newBubble.onclick = function () { toggleVote(newBubble); };

        bubbleContainer.insertBefore(newBubble, bubbleContainer.lastElementChild);

      }

    }

  </script>
